# Build-a-Complete-Medical-Chatbot-with-LLMs-LangChain-Pinecone-Flask-AWS

# HOW TO RUN

# STEP 1: Clone the repository

git clone https://github.com/kushagratyagi245/Build-a-Complete-Medical-Chatbot-with-LLMs-LangChain-Pinecone-Flask-AWS.git

# STEP 2: Go to project folder

cd Build-a-Complete-Medical-Chatbot-with-LLMs-LangChain-Pinecone-Flask-AWS

# STEP 3 (OPTION 1 - Recommended): Create virtual environment (venv)

python -m venv venv

# Activate venv (Windows)

venv\Scripts\activate

# STEP 4: Install requirements (FIXED NAME)

pip install -r requirements.txt

# STEP 5: Run the project

python app.py
